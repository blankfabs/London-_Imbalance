name: Daily Backtest

on:
  push:
    branches: [ main ]
  schedule:
    # Run daily at 12:00 PM UTC (7:00 AM EST - after London session closes)
    - cron: '0 12 * * *'
  workflow_dispatch:  # Allow manual trigger

jobs:
  backtest:
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout repository
      uses: actions/checkout@v3
      with:
        token: ${{ secrets.GITHUB_TOKEN }}
    
    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.11'
        cache: 'pip'
    
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
    
    - name: Run backtest
      run: |
        python backtest_runner.py
    
    - name: Upload results as artifact
      uses: actions/upload-artifact@v3
      with:
        name: backtest-results-${{ github.run_number }}
        path: |
          backtest_trades.csv
          backtest_metrics.json
          README.md
        retention-days: 90
    
    - name: Commit updated results
      run: |
        git config --local user.email "github-actions[bot]@users.noreply.github.com"
        git config --local user.name "github-actions[bot]"
        git add README.md backtest_trades.csv backtest_metrics.json data/eurusd_5m.csv
        git diff --quiet && git diff --staged --quiet || git commit -m "ðŸ¤– Update backtest results - Run #${{ github.run_number }}"
    
    - name: Push changes
      uses: ad-m/github-push-action@master
      with:
        github_token: ${{ secrets.GITHUB_TOKEN }}
        branch: ${{ github.ref }}
    
    - name: Create summary
      if: always()
      run: |
        echo "## Backtest Summary" >> $GITHUB_STEP_SUMMARY
        echo "" >> $GITHUB_STEP_SUMMARY
        if [ -f backtest_metrics.json ]; then
          echo "### Results" >> $GITHUB_STEP_SUMMARY
          echo '```json' >> $GITHUB_STEP_SUMMARY
          cat backtest_metrics.json >> $GITHUB_STEP_SUMMARY
          echo '```' >> $GITHUB_STEP_SUMMARY
        else
          echo "âš ï¸ No results file generated" >> $GITHUB_STEP_SUMMARY
        fi
